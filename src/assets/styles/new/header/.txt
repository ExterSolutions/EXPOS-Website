import React, { useRef, useEffect, useState } from "react";
import { Link, NavLink, useLocation } from "react-router-dom"; 
import { LuShoppingCart, LuChevronDown } from "react-icons/lu";
import "../../assets/styles/new/header/navmenu.css";

const NavMenu = ({
  dropMenu,
  handleDrawerToggle,
  location,
  isAuthenticated,
  cart,
  handleCartBarToggle,
  hideCart = false,
  scrolled = false,
}) => {
  const menuRef = useRef(null);
  const [isMobile, setIsMobile] = useState(window.innerWidth < 768);
  const [openSubmenu, setOpenSubmenu] = useState(null);
  const currentLocation = useLocation(); 

  useEffect(() => {
    const handleResize = () => {
      setIsMobile(window.innerWidth < 768);
    };
    window.addEventListener("resize", handleResize);
    return () => window.removeEventListener("resize", handleResize);
  }, []);

  // Handle click outside to close drawer
  useEffect(() => {
    const handleClickOutside = (event) => {
      if (
        dropMenu &&
        menuRef.current &&
        !menuRef.current.contains(event.target) &&
        !event.target.closest(".navbar-toggler")
      ) {
        handleDrawerToggle();
        setOpenSubmenu(null); // Close submenu
      }
    };
    if (dropMenu) {
      document.addEventListener("mousedown", handleClickOutside);
      document.body.classList.add("mobile-menu-open");
    } else {
      document.removeEventListener("mousedown", handleClickOutside);
      document.body.classList.remove("mobile-menu-open");
    }
    return () => {
      document.removeEventListener("mousedown", handleClickOutside);
      document.body.classList.remove("mobile-menu-open");
    };
  }, [dropMenu, handleDrawerToggle]);

  // Close drawer on link click (mobile)
  const handleLinkClick = () => {
    if (dropMenu) {
      handleDrawerToggle();
      setOpenSubmenu(null); // Close submenu when any link is clicked
    }
  };

  // Toggle submenu
  const handleSubmenuToggle = (menuName, event) => {
    if (event) {
      event.preventDefault();
      event.stopPropagation();
    }
    
    if (openSubmenu === menuName) {
      setOpenSubmenu(null);
    } else {
      setOpenSubmenu(menuName);
    }
  };

  // Close submenu when clicking outside (desktop hover mode)
  useEffect(() => {
    const handleClickOutsideSubmenu = (event) => {
      if (!isMobile && openSubmenu && !event.target.closest('.has-submenu')) {
        setOpenSubmenu(null);
      }
    };

    document.addEventListener('click', handleClickOutsideSubmenu);
    return () => {
      document.removeEventListener('click', handleClickOutsideSubmenu);
    };
  }, [openSubmenu, isMobile]);

  // Dynamic class for links
  const linkClass = ({ isActive }) =>
    `nav-link ${scrolled ? "nav-link-scrolled" : ""} ${
      isActive ? "active" : ""
    }`;

  // Dynamic class for ul
  const ulClass = `navbar-nav nav-menu-ul ${
    scrolled ? "justify-content-center nav-menu-scroll" : "justify-content-center"
  }`;

  const activeLocation = location || currentLocation;

  // Sample menu data with submenus (5 items as per your data, can adjust to 7)
  const menuItems = [
    { 
      name: "Home", 
      path: "/", 
      end: true 
    },
    { 
      name: "Menu", 
      path: "/menu",
      submenu: [
        { name: "Signature Pizza", path: "/signaturepizza" },
        { name: "Create Your Own", path: "/create-your-own" },
        { name: "Sides", path: "/sides" },
        { name: "Dips", path: "/dips" },
        { name: "Drinks", path: "/drinks" },
        { name: "Deals", path: "/deals" },
        { name: "Specials Offer", path: "/special-offer" },
       
      ]
    },
    { 
      name: "Stores", 
      path: "/stores" 
    },
    { 
      name: "About", 
      path: "/about-us" 
    },
    { 
      name: "Contact", 
      path: "/contact-us" 
    },
    { 
      name: isAuthenticated ? "My Account" : "Login", 
      path: isAuthenticated ? "/my-account" : "/login" 
    }
  ];

  return (
    <ul ref={menuRef} className={ulClass}>
      {menuItems.map((item, index) => (
        <li 
          key={index} 
          className={`nav-item ${item.submenu ? 'has-submenu' : ''} ${
            openSubmenu === item.name ? 'show-submenu' : ''
          }`}
          onMouseLeave={() => !isMobile && setOpenSubmenu(null)} // Close on mouse leave desktop
        >
          {item.submenu ? (
            // Menu item with submenu
            <>
              <a
                className={`nav-link ${scrolled ? "nav-link-scrolled" : ""} ${
                  activeLocation.pathname.startsWith('/menu') ? "active" : ""
                }`}
                href="#"
                onClick={(e) => handleSubmenuToggle(item.name, e)}
                onMouseEnter={() => !isMobile && setOpenSubmenu(item.name)}
                aria-haspopup="true"
                aria-expanded={openSubmenu === item.name}
              >
                {item.name}
                <LuChevronDown className="submenu-arrow" size={12} />
              </a>
              
              {/* Submenu */}
              <ul className="submenu">
                {item.submenu.map((subItem, subIndex) => (
                  <li key={subIndex}>
                    <NavLink
                      className={linkClass}
                      to={subItem.path}
                      onClick={handleLinkClick}
                    >
                      {subItem.name}
                    </NavLink>
                  </li>
                ))}
              </ul>
            </>
          ) : (
            // Regular menu item
            <NavLink
              className={linkClass}
              to={item.path}
              onClick={handleLinkClick}
              end={item.end}
            >
              {item.name}
            </NavLink>
          )}
        </li>
      ))}
      
      {/* Cart Link */}
      {!hideCart && (
        <li className="nav-item desktop-flex">
          <Link
            type="button"
            onClick={(e) => {
              e.preventDefault();
              handleCartBarToggle();
              if (dropMenu) handleDrawerToggle();
            }}
            className={`nav-link cart-link ${
              scrolled ? "nav-link-scrolled" : ""
            }`}
          >
            <LuShoppingCart size={18} />
            <span className="nav-price ms-1">
              ${cart?.grandtotal
                ? parseFloat(cart.grandtotal).toFixed(2)
                : (0.0).toFixed(2)}
            </span>
          </Link>
        </li>
      )}
    </ul>
  );
};

export default NavMenu;
